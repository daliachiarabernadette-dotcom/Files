SELECT Cognome, Nome, Data_n
FROM Pazienti, Ricoveri, Reparti
WHERE (Pazienti.cod_fiscale=Ricoveri.cod_fiscale_fk) 
and (Reparti.nome_reparto=Ricoveri.nome_reparto_fk) 
and (Reparti.nome_reparto=[Inserisci il nome del reparto]);

SELECT nome_reparto
FROM Reparti, Ricoveri, Pazienti
WHERE (Pazienti.cod_fiscale=Ricoveri.cod_fiscale_fk) 
and (Reparti.nome_reparto=Ricoveri.nome_reparto_fk)
and (Pazienti.cognome=[Inserisci il cognome])
and (Pazienti.nome=[Inserisci il nome])
and (Pazienti.data_n=[Inserisci la data di nascita]);
